Reverse a String:

#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    string str;
    cout << "Enter a string: ";
    cin >> str;

    reverse(str.begin(), str.end());  // STL function to reverse string

    cout << "Reversed String: " << str << endl;
    return 0;
}


PrimeNumber:



#include<iostream>
// #include<cstdio>
using namespace std;

int main(){
    int n;
    int c=0;
    cout<<"Enter the number";
    cin>>n;
    if (n<=1){
        cout<<"Not Prime";
        
    }
    else{
        for(int i=2;i<n;i++){
            if (n%i==0){
                c=c+1;
            };
        }
        if (c==0){
            cout<<"Prime";
            
        }
        else{
            cout<<"Not Prime";
        }
    }
    
}
